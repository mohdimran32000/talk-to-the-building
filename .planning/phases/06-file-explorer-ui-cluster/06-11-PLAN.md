Near the top of `frontend/e2e/full-suite.spec.ts` (after the existing TEST_EMAIL / TEST_PASSWORD constants and signIn helper), add two small fixture helpers. They consume the JWT stored by Supabase auth in the page's localStorage / sessionStorage so we can call backend endpoints from a Playwright test without re-authenticating.

```ts
import type { Page } from '@playwright/test'

const API_BASE = process.env.PLAYWRIGHT_API_BASE ?? 'http://localhost:8001'

/**
 * Read the Supabase access token from the page's storage. Supabase v2 stores it under
 * `sb-<projectRef>-auth-token` in localStorage; the project ref varies, so we scan all
 * localStorage keys for one starting with `sb-` and ending in `-auth-token`.
 * Returns null if the user is not signed in.
 */
async function getStoredToken(page: Page): Promise<string | null> {
  return await page.evaluate(() => {
    for (const key of Object.keys(window.localStorage)) {
      if (key.startsWith('sb-') && key.endsWith('-auth-token')) {
        try {
          const raw = window.localStorage.getItem(key)
          if (!raw) continue
          const parsed = JSON.parse(raw)
          return parsed?.access_token ?? null
        } catch {
          continue
        }
      }
    }
    return null
  })
}

/**
 * apiPost — fixture helper for tests that need to bootstrap folder structure without
 * 5 UI clicks. Reused by Tests 12 + 13 (per checker WARNING #4). The signed-in
 * page must call this; getStoredToken() reads the token Supabase put in localStorage
 * during signIn(page).
 */
async function apiPost(page: Page, path: string, body: unknown): Promise<any> {
  const token = await getStoredToken(page)
  if (!token) throw new Error('apiPost: no auth token in localStorage — call signIn(page) first')
  const res = await page.request.post(`${API_BASE}${path}`, {
    headers: { Authorization: `Bearer ${token}`, 'Content-Type': 'application/json' },
    data: body,
  })
  if (!res.ok()) {
    throw new Error(`apiPost ${path} failed: ${res.status()} ${await res.text()}`)
  }
  return res.json()
}

/**
 * apiDelete — fixture cleanup helper. Used in `finally` blocks to remove
 * test-created folders/documents by id. CLAUDE.md mandatory rule: tests
 * must NEVER delete all user data — pass specific ids only.
 */
async function apiDelete(page: Page, path: string): Promise<void> {
  const token = await getStoredToken(page)
  if (!token) throw new Error('apiDelete: no auth token in localStorage')
  const res = await page.request.delete(`${API_BASE}${path}`, {
    headers: { Authorization: `Bearer ${token}` },
  })
  // Tolerate 404 in cleanup (already gone)
  if (!res.ok() && res.status() !== 404) {
    console.warn(`apiDelete ${path} returned ${res.status()}: ${await res.text()}`)
  }
}
```
