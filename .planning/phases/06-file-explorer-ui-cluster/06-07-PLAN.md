```tsx
  Folder,

} from 'lucide-react'

const EXPLORER_TOOL_ICON: Record<string, React.ComponentType<{ className?: string }>> = {
  tree: FolderTreeIcon,
  list_files: Folder,
  glob: FileSearch,
  grep: Search,
  read_document: Eye,
  search_documents: Search,
  analyze_document: FileText,

function summarizeArgs(args: Record<string, unknown> | undefined): string {
  if (!args) return ''
  const arg = (args.path ?? args.pattern ?? args.query ?? args.document_name ?? args.question) as string | undefined
  return typeof arg === 'string' ? arg : ''
}

export interface ToolCallRowProps {
  call: ToolCallEntry
}

export function ToolCallRow({ call }: ToolCallRowProps) {
  const Icon = EXPLORER_TOOL_ICON[call.tool] ?? Search
  const summary = summarizeArgs(call.args as Record<string, unknown> | undefined)
  return (
    <div className="ml-4 flex items-center gap-2 text-xs text-muted-foreground">
      <Icon className="w-3 h-3 shrink-0" />
      <span className="font-mono shrink-0">{call.tool}</span>
      {summary && <span className="italic truncate" title={summary}>{summary}</span>}
      {typeof call.turn === 'number' && (
        <span className="ml-auto shrink-0 text-muted-foreground/60">turn {call.turn}</span>
      )}
    </div>
  )
}
```

```tsx
import type { ToolUsedEntry } from '@/lib/api'
import { ToolCallRow } from './ToolActivity'
import { useState, useMemo } from 'react'
import MarkdownContent from './MarkdownContent'  // existing import — keep the path the file already uses

interface SubAgentSectionProps {
  tool: ToolUsedEntry
  isLive?: boolean
  defaultExpanded?: boolean
}

function SubAgentSection({ tool, isLive, defaultExpanded = false }: SubAgentSectionProps) {
  const [expanded, setExpanded] = useState(isLive ?? defaultExpanded)
  // Presentation-only string formatting (NOT a behavior fork — Pitfall 12 allows this)
  const label = useMemo(() => {
    if (tool.tool === 'analyze_document') {
      return isLive ? `Analyzing "${tool.document_name}"...` : `Analyzed "${tool.document_name}"`
    }
    if (tool.tool === 'explore_knowledge_base') {
      return isLive ? `Exploring: ${tool.question}` : `Explored: ${tool.question}`
    }
    return isLive ? `Running ${tool.tool}` : `Used ${tool.tool}`
  }, [tool, isLive])
  return (
    <div className="border-l-2 border-blue-400/50 pl-3 ml-1 my-2">
      <button
        type="button"
        onClick={() => setExpanded(!expanded)}
        className="flex items-center gap-1 text-xs text-blue-400 hover:text-blue-300 transition-colors"
      >
        <span className="font-mono">{expanded ? '▼' : '▶'}</span>
        <span>{label}</span>
        {isLive && <span className="animate-pulse ml-1">●</span>}
      </button>
      {expanded && (
        <>
          {/* Recursion seam — empty array for analyze_document, populated for explore_knowledge_base. NO if-branch. */}
          {tool.tool_calls && tool.tool_calls.length > 0 && (
            <div className="mt-1 space-y-1">
              {tool.tool_calls.map((call, i) => (
                <ToolCallRow key={i} call={call} />
              ))}
            </div>
          )}
          {tool.sub_agent_result && (
            <div className="mt-1 text-xs opacity-80">
              <MarkdownContent content={tool.sub_agent_result} />
            </div>
          )}
        </>
      )}
    </div>
  )
}
```

```tsx
onSubAgentStart: (payload) => {
  // payload: { sub_agent_id, agent_name, document_name?, question? }
  setLiveSubAgentTrace({
    tool: payload.agent_name,                          // 'analyze_document' | 'explore_knowledge_base'
    sub_agent_id: payload.sub_agent_id,
    tool_calls: [],
    document_name: payload.document_name,
    question: payload.question,
    sub_agent_result: '',
  })
},
onSubAgentToken: (chunk) => {
  setLiveSubAgentTrace((prev) =>
    prev ? { ...prev, sub_agent_result: (prev.sub_agent_result ?? '') + chunk } : prev
  )
},
onSubAgentToolStart: (payload) => {
  // payload: { tool, args, turn }
  setLiveSubAgentTrace((prev) => {
    if (!prev) return prev
    const newCall: ToolCallEntry = { tool: payload.tool, args: payload.args, turn: payload.turn, status: 'running' }
    return { ...prev, tool_calls: [...(prev.tool_calls ?? []), newCall] }
  })
},
onSubAgentToolDone: (payload) => {
  // payload: { tool, result_preview, turn }
  setLiveSubAgentTrace((prev) => {
    if (!prev) return prev
    const updated = (prev.tool_calls ?? []).map((c) =>
      c.turn === payload.turn && c.tool === payload.tool
        ? { ...c, result_preview: payload.result_preview, status: 'done' as const }
        : c
    )
    return { ...prev, tool_calls: updated }
  })
},
onSubAgentDone: () => {
  // The persisted message will rehydrate via tool_metadata; clear live trace once stream completes
  setLiveSubAgentTrace(null)
},
```
