**2) Pitfall-5-aware `deleteFolder`** — does NOT use `fetchApi` because the helper throws on `!res.ok` and would lose the structured 409 body. Verbatim from 06-PATTERNS.md lines 393-409 / RESEARCH.md §Pitfall 5:

```ts
export type DeleteFolderResult =
  | { ok: true }
  | { ok: false; error: 'FOLDER_NOT_EMPTY'; document_count: number; subfolder_count: number }

export async function deleteFolder(id: string): Promise<DeleteFolderResult> {
  const token = await getToken()
  const res = await fetch(`/api/folders/${id}`, {
    method: 'DELETE',
    headers: { Authorization: `Bearer ${token}` },
  })
  if (res.status === 200) return { ok: true }
  if (res.status === 409) {
    const body = await res.json()
    return {
      ok: false,
      error: body.error,           // 'FOLDER_NOT_EMPTY' per backend contract
      document_count: body.document_count,
      subfolder_count: body.subfolder_count,
    }
  }
  const err = await res.json().catch(() => ({}))
  throw new Error(err.detail || `Delete failed: ${res.status}`)
}
```

**3) Extend the existing `uploadFile`** to accept `folder_path` + `scope` query args. Read the current `uploadFile` signature first — typically `(file: File): Promise<UploadedFile>`. Extend to:

```ts
export async function uploadFile(
  file: File,
  folder_path: string = '/',
  scope: 'user' | 'global' = 'user'
): Promise<UploadedFile> {
  const token = await getToken()
  const formData = new FormData()
  formData.append('file', file)
  const qs = new URLSearchParams({ folder_path, scope }).toString()
  const res = await fetch(`/api/files/upload?${qs}`, {
    method: 'POST',
    headers: { Authorization: `Bearer ${token}` },     // NO Content-Type — let browser set multipart boundary
    body: formData,
  })
  if (!res.ok) {
    const body = await res.json().catch(() => ({}))
    throw new Error(body.detail || `Upload failed: ${res.status}`)
  }
  return res.json()
}
```
